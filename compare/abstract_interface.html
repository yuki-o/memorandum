Objective-C では、インターフェースではなくプロトコルと呼んでいる。 @protocol を使用して定義する。 メンバメソッドの実装を必須にするには @required を、実装を任意にするには @optional を指定する。 何も指定しない場合は @required 扱いとなる。 プロトコルはメンバ変数を持つことができない。 2. プロトコルをクラスで採用する場合は <プロトコル名> を付ける。 3. プロトコルの実装は、そのプロトコルを採用したクラスで行う。 4. Objective-C には抽象クラスがないので、インスタンス化ができてしまう。 継承先でメンバメソッドの振る舞いを決めたい場合は、基底のメンバメソッドに例外を仕込んでおくのが良さそう。 5. プロトコルは、インスタンス化することができない. __interface.h__ ```objectivec // (1) インターフェース (生物). @protocol Creature @required // 逃げる (生物によって逃げ方が違う). -(void) runAway; @optional @end /* @protocol Creature */ //--------------------------------------- // (1) インターフェース (人間). @protocol Human @required // 自己紹介をする. -(void) introduceOneself; // 仕事をする (人によって仕事の内容が違う). -(void) work; @end /* @protocol Human */ ``` __human.h__ ```objectivec #import <Foundation/Foundation.h> // クラス (人間) の宣言. @interface HumanBase : NSObject <Creature, Human>　 // (2). { @protected NSString* name; // 名前. } -(id) initWithName:(NSString*)i_name; @end /* @interface HumanBase */ //--------------------------------------- // クラス (人間) の実装. @implementation HumanBase // イニシャライザ. -(id) initWithName:(NSString*)i_name { if( self = [super init] ){ name = i_name; } return self; } // (3) 逃げる. -(void) runAway { NSLog( @"走って逃げる" ); } // 自己紹介をする. -(void) introduceOneself { NSLog( @"私の名前は%@です", name ); } // (4) 仕事をする (人によって仕事の内容が違う). -(void) work { // 継承したクラスで実装する.
