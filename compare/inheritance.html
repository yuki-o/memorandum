## ~ 継承、オーバーライド ~ ### C言語 * なし --- ### C++ 1. 基底クラスのデストラクタと、オーバーライドしたいメンバメソッドに virtual を付ける。 ただし (7) のように、基底クラスのポインタ型変数に派生クラス型のインスタンスを代入するといった使い方をしない場合は付ける必要がない。 2. クラス内のメンバにアクセスする際に、this ポインタを使用してもよい。 3. 継承修飾子を用いて継承する。継承修飾子については下記の表を参照のこと。 継承修飾子を指定しない場合、構造体では public 扱い、クラスでは private 扱いとなる。 ただし、継承修飾子に関わらず、派生クラスからは基底クラスの protected と public なメンバにアクセスすることができる。 4. 派生クラスのコンストラクタから、「基底クラス名( 引数 )」という書き方で、基底クラスのコンストラクタを呼び出すことができる。 ただし、コンストラクタ初期化子 : の部分でしか呼び出せない。 5. 基底クラスと同じメンバメソッドを派生クラスで再定義することで、そのメンバメソッドをオーバーライドすることができる。 ただし、メンバ変数はオーバーライドすることができない。 virtual は付けなくてもよいが、オーバーライドした目印として付けている。 6. 派生クラスから、「基底クラス名::メンバ」という書き方で基底クラスのメンバにアクセスすることができる。 __継承修飾子__ 修飾子 | 詳細 --- | --- private | 基底クラスのメンバを全て private にして引き継ぐ protected | 基底クラスでアクセス修飾子が public だったメンバを protected にして引き継ぐ (他はそのまま) public | そのまま引き継ぐ __aircrapt.h__ ```cpp #ifndef __aircrapt__ #define __aircrapt__ #include <iostream> using namespace std; // 基底クラス (航空機). class Aircrapt { private: int size; // 大きさ. protected: int speed; // 速度. public: // コンストラクタ. Aircrapt( int i_nSize, int i_nSpeed ) : size ( i_nSize ) , speed ( i_nSpeed ) { cout << "Aircrapt::Aircrapt" << endl; } // (1) デストラクタ. virtual ~Aircrapt() { cout << "Aircrapt::~Aircrapt" << endl; } // (1) 飛ぶ. virtual void fly() const { cout << "Aircrapt::fly" << endl; } // (1)(2) メンバ変数の値を表示. virtual void disp() const { cout << "size:" << this->size << endl; } }; #endif /* defined(__aircrapt__) */ ``` __fighter.h__ ```cpp #ifndef __fighter__ #define __fighter__ #include <iostream> #include "aircrapt.h" using namespace std; // 派生クラス (戦闘機). class Fighter : public Aircrapt // (3) 継承. { private: int bulletNum; // 弾の数. public: // コンストラクタ. Fighter( int i_nSize, int i_nSpeed, int i_nBulletNum ) : Aircrapt ( i_nSize, i_nSpeed ) // (4) 基底クラスのコンストラクタの呼び出し. , bulletNum ( i_nBulletNum ) { cout << "Fighter::Fighter" << endl; } // デストラクタ. virtual ~Fighter() { cout << "Fighter::~Fighter" << endl; } // (5) 飛ぶ. virtual void fly() const { cout << "Fighter::fly" << endl; } // (5) メンバ変数の値を表示. virtual void disp() const { Aircrapt::disp(); // (6) 基底クラスの disp の呼び出し. cout << "speed:" << Aircrapt::speed << endl; // (6) 基底クラスの speed にアクセス. cout << "bulletNum:" << bulletNum << endl; } }; #endif /* defined(__fighter__) */ ``` __main.cpp__ ```cpp #include <iostream> #include "aircrapt.h" #include "fighter.h" using namespace std; int main( int argc, const char * argv[] ) { // Fighter のポインタ型変数に、Fighter 型のインスタンスを代入. Fighter* f1 = new Fighter( 10, 20, 30 ); cout << "---(a)---" << endl; f1->fly(); cout << "---(b)---" << endl; f1->disp(); cout << "---(c)---" << endl; delete f1; cout << "\n===\n" << endl; // (7) Aircrapt のポインタ型変数に、Fighter 型のインスタンスを代入. Aircrapt* f2 = new Fighter( 10, 20, 30 ); cout << "---(d)---" << endl; f2->fly(); cout << "---(e)---" << endl; f2->disp(); cout << "---(f)---" << endl; delete f2; return 0; } ``` __出力結果__ ``` Aircrapt::Aircrapt Fighter::Fighter ---(a)--- Fighter::fly ---(b)--- size:10 speed:20 bulletNum:30 ---(c)--- Fighter::~Fighter Aircrapt::~Aircrapt === Aircrapt::Aircrapt Fighter::Fighter ---(d)--- Fighter::fly ---(e)--- size:10 speed:20 bulletNum:30 ---(f)--- Fighter::~Fighter Aircrapt::~Aircrapt ``` --- ### Java 1. クラス内のメンバにアクセスする際に、this を使用してもよい。 2. extends 修飾子を用いて継承する。 3. 派生クラスのコンストラクタから、「super( 引数 );」という書き方で、基底クラスのコンストラクタを呼び出すことができる。 ただし、コンストラクタの1行目でしか呼び出せない。 4. 基底クラスと同じメンバメソッドを派生クラスで再定義することで、そのメンバメソッドをオーバーライドすることができる。 ただし、メンバ変数はオーバーライドすることができない。 @Override は付けなくてもよいが、付けた方がオーバーライドを明示することができるし、オーバーライドに失敗している場合はコンパイルエラーとなる。 5. 派生クラスから、「super.メンバ」という書き方で基底クラスのメンバにアクセスすることができる。 __Aircrapt.java__ ```java // 基底クラス (航空機). class Aircrapt { private int size; // 大きさ. protected int speed; // 速度. // コンストラクタ. public Aircrapt( int i_nSize, int i_nSpeed ) { size = i_nSize; speed = i_nSpeed; System.out.println( "Aircrapt::Aircrapt" ); } // 飛ぶ. public void fly() { System.out.println( "Aircrapt::fly" ); } // (1) メンバ変数の値を表示. public void disp() { System.out.println( "size:" + this.size ); } } ``` __Fighter.java__ ```java // 派生クラス (戦闘機). class Fighter extends Aircrapt // (2) 継承. { private int bulletNum; // 弾の数. // コンストラクタ. public Fighter( int i_nSize, int i_nSpeed, int i_nBulletNum ) { super( i_nSize, i_nSpeed ); // (3) 基底クラスのコンストラクタの呼び出し. bulletNum = i_nBulletNum; System.out.println( "Fighter::Fighter" ); } // (4) 飛ぶ. @Override public void fly() { System.out.println( "Fighter::fly" ); } // (4) メンバ変数の値を表示. @Override public void disp() { super.disp(); // (5) 基底クラスの disp の呼び出し. System.out.println( "speed:" + super.speed ); // (5) 基底クラスの speed にアクセス. System.out.println( "bulletNum:" + bulletNum ); } } ``` __Main.java__ ```java class Main { public static void main( String[] args ) { // Fighter 型変数に、Fighter 型のインスタンスを代入. Fighter f1 = new Fighter( 10, 20, 30 ); System.out.println( "---(a)---" ); f1.fly(); System.out.println( "---(b)---" ); f1.disp(); System.out.println( "---(c)---" ); System.out.println( "\n===\n" ); // Aircrapt 型変数に、Fighter 型のインスタンスを代入. Aircrapt f2 = new Fighter( 10, 20, 30 ); System.out.println( "---(d)---" ); f2.fly(); System.out.println( "---(e)---" ); f2.disp(); System.out.println( "---(f)---" ); } } ``` __出力結果__ ``` Aircrapt::Aircrapt Fighter::Fighter ---(a)--- Fighter::fly ---(b)--- size:10 speed:20 bulletNum:30 ---(c)--- === Aircrapt::Aircrapt Fighter::Fighter ---(d)--- Fighter::fly ---(e)--- size:10 speed:20 bulletNum:30 ---(f)--- ``` --- ### C# # 1. 基底クラスのオーバーライドしたいメンバメソッドに virtual を付ける。 2. クラス内のメンバにアクセスする際に、this を使用してもよい。 3. 派生クラスのコンストラクタから、「: base( 引数 )」という書き方で、基底クラスのコンストラクタを呼び出すことができる。 ただし、サンプルプログラムに示すように { の直前にしか書けない。 4. 基底クラスと同じメンバメソッドを派生クラスで再定義し、override を付けることで、そのメンバメソッドをオーバーライドすることができる。 ただし、メンバ変数はオーバーライドすることができない。 5. 派生クラスから、「base.メンバ」という書き方で基底クラスのメンバにアクセスすることができる。 __Aircrapt.cs__ ```csharp using System; // 基底クラス (航空機). class Aircrapt { private int size; // 大きさ. protected int speed; // 速度. // コンストラクタ. public Aircrapt( int i_nSize, int i_nSpeed ) { size = i_nSize; speed = i_nSpeed; Console.WriteLine( "Aircrapt::Aircrapt" ); } // (1) 飛ぶ. public virtual void fly() { Console.WriteLine( "Aircrapt::fly" ); } // (1)(2) メンバ変数の値を表示. public virtual void disp() { Console.WriteLine( "size:" + this.size ); } } ``` __Fighter.cs__ ```csharp using System; // 派生クラス (戦闘機). class Fighter : Aircrapt // 継承. { private int bulletNum; // 弾の数. // コンストラクタ. public Fighter( int i_nSize, int i_nSpeed, int i_nBulletNum ) : base( i_nSize, i_nSpeed ) // (3) 基底クラスのコンストラクタの呼び出し. { bulletNum = i_nBulletNum; Console.WriteLine( "Fighter::Fighter" ); } // (4) 飛ぶ. public override void fly() { Console.WriteLine( "Fighter::fly" ); } // (4) メンバ変数の値を表示. public override void disp() { base.disp(); // (5) 基底クラスの disp の呼び出し. Console.WriteLine( "bulletNum:" + bulletNum ); } } ``` __Program.cs__ ```csharp using System; class Program { public static void Main( string[] args ) { // Fighter 型変数に、Fighter 型のインスタンスを代入.
